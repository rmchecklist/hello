git clone repository ==> git clone {url}
add file ==> git add <filename>
commit file ==> git commit -m "message"
git status ==> check the status of the local/repository
git push ==> push the changes to repo
git commit -am "message" ==> add and commit files"
git pull ==> pull the changes from repo
git log ==> read all commit log info
git reset --hard {commit hash}
git brach ==> return the current working branch
git checkout -b style ==> create a new branch
git checkout master ==> switch branch
git merge style
